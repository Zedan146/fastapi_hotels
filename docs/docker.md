  # –í–µ–±-–ø—Ä–∏–ª–æ–∂–µ–Ω–∏–µ

      context: .
      dockerfile: Dockerfile
    ports:
      - "8000:8000"

      - REDIS_URL=redis://redis:6379/0
      - MODE=PROD

      db:
        condition: service_healthy
      redis:
        condition: service_started
    volumes:
      - ./logs:/app/logs
    restart: unless-stopped

      - app-network

  # –ë–∞–∑–∞ –¥–∞–Ω–Ω—ã—Ö

      - postgres_data:/var/lib/postgresql/data

    ports:
      - "5432:5432"
    healthcheck:
      test: ["CMD-SHELL", "pg_isready -U user -d hotels_db"]
      interval: 10s
      timeout: 5s
      retries: 5
    restart: unless-stopped
    networks:
      - app-network

  # Redis

    image: redis:7-alpine
    ports:
      - "6379:6379"
    volumes:
      - redis_data:/data
    command: redis-server --appendonly yes
    restart: unless-stopped
    networks:
      - app-network

    build: .
    command: celery -A src.tasks.celery_app:celery_instance worker --loglevel=info

      - DATABASE_URL=postgresql://user:password@db:5432/hotels_db

    depends_on:
      - db
      - redis
    volumes:
      - ./logs:/app/logs
    restart: unless-stopped
    networks:
      - app-network

    build: .

      - DATABASE_URL=postgresql://user:password@db:5432/hotels_db

    depends_on:
      - db
      - redis
    volumes:
      - ./logs:/app/logs
    restart: unless-stopped
    networks:
      - app-network

    image: nginx:alpine
    ports:
      - "80:80"
      - "443:443"
    volumes:
      - ./nginx.conf:/etc/nginx/nginx.conf
      - ./ssl:/etc/nginx/ssl
    depends_on:
      - web
    restart: unless-stopped

      - app-network

volumes:
  postgres_data:
  redis_data:
